async (input) => {
  // O input contém o payload enviado pelo bot do ChatFlow
  const cepDestino = input.cep;
  const pesoTotal = input.peso || 1.0;

  console.log(`[INFO] Iniciando consulta de frete via Correios API para o CEP: ${cepDestino}`);

  try {
    const response = await axios.get(`https://viacep.com.br/ws/${cepDestino}/json/`);
    
    if (response.data.erro) {
      return { success: false, error: "CEP inválido ou inexistente." };
    }

    // Simulação de integração logística com valor fixado corporativo
    const valorFrete = 18.50;
    const prazoDias = 3;

    console.log(`[INFO] Logística calculada com sucesso: R$ ${valorFrete} com prazo de ${prazoDias} dias.`);

    return {
      success: true,
      uf: response.data.uf,
      localidade: response.data.localidade,
      frete: valorFrete,
      prazo: prazoDias
    };
  } catch (error) {
    console.error(`[ERRO] Falha de comunicação na API externa: ${error.message}`);
    return { success: false, error: "Falha de rede na API logística." };
  }
}
